[tex] \frac{x1}{y2} = \frac{x2}{y1} \\ = \frac{9}{y2} = \frac{12}{20} \\ = \frac{9 \times 20}{12} = \frac{180}{12} = 15 \: hari[/tex]

[tex] \frac{x1}{y1} = \frac{x2}{y2} \\ = \frac{5}{60} = \frac{40}{y2} \\ = \frac{60 \times 40}{5} = \frac{2400}{5} = 480 \: km[/tex]
